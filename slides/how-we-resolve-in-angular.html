How we resolve in Angular


@Injectable()
export class companyWorkersResolver implements Resolve{
    constructor(private companyWorkersService: companyWorkersService) {}
    resolve(route: ActivatedRouteSnapshot) {
        return companyWorkersService.getWorkers(route.queryParams.companyId);
    }
}

@NgModule({
    providers: [
        companyWorkersResolver
    ],
    imports: [
        RouterModule.forRoot([{
            ...
            resolve: {
                companyWorkers: companyWorkersResolver
            }
        }])
    ]
})
export class WAEditDetailsModule {}
