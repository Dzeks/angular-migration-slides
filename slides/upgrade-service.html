Upgrade & downgrade ng1 service


@Injectable()
export class CompanyService {
    companyId: string;
    constructor(private api: ApiService) {}
    getHistory(companyId) {
        return this.api.get(`/api/companies/${companyId}/history`);
    }
}
export let companyDomain = angular.module('companyDomain', [])
    .service('companyService', downgradeInjectable(CompanyService))
    .name;
