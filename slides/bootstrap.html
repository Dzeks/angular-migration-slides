"When you use ngUpgrade, what you're really doing is running both AngularJS and Angular at the same time." Link 


platformBrowserDynamic().bootstrapModule(AppModule).then(platformRef => {
    const injector: Injector = platformRef.injector;
    const upgrade = injector.get(UpgradeModule) as UpgradeModule;
    upgrade.bootstrap(document.body, [application], {strictDi: true});
});
